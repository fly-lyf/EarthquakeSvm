function showDialogSize(url,width,height){
	var availHeight = window.screen.availHeight;
	var availWidth = window.screen.availWidth;
	// 大致居中
	var iWidth = width - 10; // 窗口宽度
	var iHeight = height - 30;// 窗口高度
	var iLeft = (availWidth - width) / 2;
	var iTop = (availHeight - height) / 2;
	var newWindow1 = window.open(url, "","scrollbars=yes,toolbar=no,location=no,Direction=no,resizable=no,Width="+ iWidth + ",Height=" + iHeight + ",top=" + iTop + ",left="+ iLeft);
	newWindow1.focus();// 窗口获取焦点
}

<P>　　<SPAN>11</SPAN><SPAN>月<SPAN>9</SPAN>日，雅安市天全县防震减灾局组织召开<SPAN>2012</SPAN>年度防震减灾工作总结暨《四川省防震减灾条例》培训会。天全县副县长黄敏、应急办主任陈洪、全县<SPAN>15</SPAN>个乡镇防震减灾工作分管领导及助理员、城厢镇各社区主任共计<SPAN>30</SPAN>余人参加了会议。</SPAN></P>
<P>　　<SPAN>会议结合雅安市对该县的目标考核，对<SPAN>2012</SPAN>年全市防震减灾各项工作进行了认真总结，指出了工作中暴露出的薄弱环节和突出问题；传达学习了<SPAN>2012</SPAN>年<SPAN>10</SPAN>月<SPAN>1</SPAN>日起开始实施的《四川省防震减灾条例》；共同交流了防震减灾助理员日常业务工作开展中较好的经验和存在的问题。最后，黄敏对防震减灾工作提出了三点要求：<SPAN>一是抓好防震减灾工作机构队伍建设，全面落实防震减灾责任制；二是抓好建设工程抗震设防要求管理，切实提高震害防御能力；三是强化防震减灾宣传教育，增强全社会防震减灾意识和能力。</SPAN></SPAN></P>
