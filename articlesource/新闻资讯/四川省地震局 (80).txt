function showDialogSize(url,width,height){
	var availHeight = window.screen.availHeight;
	var availWidth = window.screen.availWidth;
	// 大致居中
	var iWidth = width - 10; // 窗口宽度
	var iHeight = height - 30;// 窗口高度
	var iLeft = (availWidth - width) / 2;
	var iTop = (availHeight - height) / 2;
	var newWindow1 = window.open(url, "","scrollbars=yes,toolbar=no,location=no,Direction=no,resizable=no,Width="+ iWidth + ",Height=" + iHeight + ",top=" + iTop + ",left="+ iLeft);
	newWindow1.focus();// 窗口获取焦点
}

<P>　　<SPAN>&nbsp;<SPAN>2016</SPAN><SPAN>年<SPAN>1</SPAN>月<SPAN>6</SPAN>日，阿坝州政府召开全州防震减灾工作领导小组会议，传达贯彻落实省防震减灾工作领导小组会议精神，总结<SPAN>2015</SPAN>年防震减灾工作，安排部署<SPAN>2016</SPAN>年重点工作任务，通报全省、全州<SPAN>2016</SPAN>年度地震活动趋势意见。州防震减灾领导小组成员单位负责人参加了会议。阿坝州政府副州长李为国出席会议并讲话。</SPAN></P>
<P>　　李为国在讲话中要求，2016年要切实加强组织领导，建立健全党委领导、政府负责、社会协同、公众参与、法治保障的防震减灾工作治理体制，落实好防震减灾“十三五”规划；要加强震情跟踪和应急准备，力争最大限度地减轻地震灾害损失；要加强地震监测预报工作，落实震情信息快速反应机制，推动地震监测技术信息化与社会需求的深度融合；要提升抗御地震灾害的能力，加强对建设工程的抗震设防要求监管，组织开展好农居抗震加固改造工作；要加强救援救助体系建设，进一步完善地震应急预案，完善应急协调机制；要强化防震减灾宣传教育，增强全民防震减灾意识。</P>
